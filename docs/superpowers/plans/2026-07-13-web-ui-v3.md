# margay ui v3 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Two-pane layout (worktree list → detail with URLs block and pre-opened log tabs) and a working :80 bind on macOS (wildcard + loopback peer guard).

**Architecture:** `lib/ui.py` gains a two-step proxy bind (`127.0.0.1` → wildcard on `PermissionError`) with a `loopback_peer()` guard at the top of every proxy request; `lib/ui.html` is rebuilt as sidebar + detail (summary with URLs block, per-worktree auto-opened log tabs below, divider resize). `/api/state` is unchanged. Spec: `docs/superpowers/specs/2026-07-13-web-ui-v3-design.md`.

**Tech Stack:** Python 3 stdlib; vanilla JS/CSS; bash test suites.

## Global Constraints

- Python 3 stdlib only. UI listener (7997) keeps its plain `127.0.0.1` bind and byte-identical `origin_ok` guard.
- Proxy bind order: `("127.0.0.1", port)`; on `PermissionError` retry `("", port)` (wildcard). Wildcard mode REQUIRES the peer guard: `loopback_peer(ip)` true only for `127.*`, `::1`, `::ffff:127.*`; non-loopback peers → 403 + connection close, checked before any routing. Both-binds-fail → existing warning + fallback (PROXY["port"] stays None).
- Env `MARGAY_UI_WILDCARD=1` (tests only) skips the loopback attempt and binds wildcard directly.
- Tests never bind :80; wildcard test uses a random high port.
- `lib/ui.html`: no framework, no external resources; `esc()` for HTML contexts, `jsq()` inside inline handlers; the 2 s poll re-renders sidebar + summary only — log panes update in place (offset append, scroll/follow preserved, 256 KiB cap).
- Tab state per worktree (open tabs incl. command, active tab, follow, buffers) survives selection switches in memory; selection and divider height persist in `localStorage` (`margaySelected`, `margayLogH`).
- All three suites green at every commit; `python3 -m py_compile lib/ui.py` clean (then `rm -rf lib/__pycache__`).

"use strict";
const esc = s => String(s).replace(/[&<>"']/g,
  c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));
const jsq = s => esc(String(s).replace(/\\/g, '\\\\').replace(/'/g, "\\u0027").replace(/"/g, '\\u0022'));

const busy = new Set();
let lastState = {projects: []};
let selected = localStorage.getItem('margaySelected') || null;

// per-worktree tab state: wtPath -> { tabs: Map(key -> tab), active: key }
// tab: {kind:'log'|'cmd', title, logPath?, offset, buf, follow, err, meta:[{text,href?}]}
const wtTabs = new Map();
function tabsFor(wt) {
  if (!wtTabs.has(wt)) wtTabs.set(wt, { tabs: new Map(), active: null });
  return wtTabs.get(wt);
}

const logarea = document.getElementById('logarea');
const logtabs = document.getElementById('logtabs');
const logpanes = document.getElementById('logpanes');
logarea.style.height = (parseInt(localStorage.getItem('margayLogH'), 10) || 240) + 'px';

function findWt(path) {
  for (const p of lastState.projects || [])
    for (const wt of p.worktrees || [])
      if (wt.path === path) return { p, wt };
  return null;
}

function defaultSelection() {
  let first = null;
  for (const p of lastState.projects || [])
    for (const wt of p.worktrees || []) {
      first = first || wt.path;
      if (wt.services.length) return wt.path;
    }
  return first;
}

function selectWt(path) {
  selected = path;
  localStorage.setItem('margaySelected', path);
  render();
}

// keep the selected worktree's tabs in sync with its running services
function syncTabs(wt) {
  const st = tabsFor(wt.path);
  const wanted = new Set(wt.services.map(s => 'log:' + s.service));
  for (const key of [...st.tabs.keys()])
    if (key.startsWith('log:') && !wanted.has(key)) {
      st.tabs.delete(key);
      const pre = logpanes.querySelector('pre[data-key="' + CSS.escape(wt.path + '|' + key) + '"]');
      if (pre) pre.remove();
    }
  for (const s of wt.services) {
    const key = 'log:' + s.service;
    if (!st.tabs.has(key) && s.log) {
      st.tabs.set(key, { kind:'log', title: s.service, logPath: s.log,
                         offset:-1, buf:'', follow:true, err:false,
                         meta: [ s.url ? {text:s.url, href:s.url} : null,
                                 {text:'pid ' + s.pid},
                                 s.dbName ? {text:'db ' + s.dbName} : null ].filter(Boolean) });
    } else if (st.tabs.has(key)) {
      const t = st.tabs.get(key);
      t.meta = [ s.url ? {text:s.url, href:s.url} : null, {text:'pid ' + s.pid},
                 s.dbName ? {text:'db ' + s.dbName} : null ].filter(Boolean);
    }
  }
  if (!st.active || !st.tabs.has(st.active)) st.active = st.tabs.keys().next().value || null;
}

function paneFor(wt, key) {
  const id = wt + '|' + key;
  let pre = logpanes.querySelector('pre[data-key="' + CSS.escape(id) + '"]');
  if (!pre) {
    pre = document.createElement('pre');
    pre.className = 'pane';
    pre.dataset.key = id;
    logpanes.appendChild(pre);
  }
  return pre;
}

function renderTabs() {
  const st = selected ? tabsFor(selected) : null;
  logtabs.innerHTML = '';
  if (!st) return;
  if (st.active) paneFor(selected, st.active);
  for (const pre of logpanes.querySelectorAll('.pane'))
    pre.classList.toggle('act', pre.dataset.key === selected + '|' + (st.active || ''));
  for (const [key, t] of st.tabs) {
    const el = document.createElement('span');
    el.className = 'dtab' + (key === st.active ? ' act' : '');
    el.innerHTML = esc(t.title) +
      (t.kind === 'cmd' ? ' <span class="close" title="close">✕</span>' : '') +
      (t.kind === 'log'
        ? ' <label style="font-size:.68rem"><input type="checkbox" ' + (t.follow ? 'checked' : '') + '> follow</label>'
        : '');
    el.addEventListener('click', e => {
      if (e.target.classList && e.target.classList.contains('close')) {
        st.tabs.delete(key);
        const pre = logpanes.querySelector('pre[data-key="' + CSS.escape(selected + '|' + key) + '"]');
        if (pre) pre.remove();
        if (st.active === key) st.active = st.tabs.keys().next().value || null;
        renderTabs(); return;
      }
      if (e.target.type === 'checkbox') { t.follow = e.target.checked; return; }
      st.active = key; renderTabs();
    });
    logtabs.appendChild(el);
  }
  const t = st.tabs.get(st.active);
  const pre = t && paneFor(selected, st.active);
  if (pre && t) { pre.textContent = t.buf; pre.classList.toggle('err', !!t.err); if (t.follow) pre.scrollTop = pre.scrollHeight; }
}

async function pollLogs() {
  if (!selected) return;
  const st = tabsFor(selected);
  for (const [key, t] of st.tabs) {
    if (t.kind !== 'log') continue;
    try {
      const r = await fetch('/api/log?file=' + encodeURIComponent(t.logPath) + '&offset=' + t.offset);
      if (r.status !== 200) continue;
      const data = await r.json();
      if (data.offset < t.offset) t.buf = data.data; else t.buf += data.data;
      if (t.buf.length > 262144) t.buf = t.buf.slice(-262144);
      t.offset = data.offset;
      const pre = paneFor(selected, key);
      if (data.data || pre.textContent === '') {
        const stick = t.follow;
        pre.textContent = t.buf;
        if (stick && key === st.active) pre.scrollTop = pre.scrollHeight;
      }
    } catch (e) { /* next tick */ }
  }
}

async function api(path, body) {
  const opts = body === undefined ? {} :
    { method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify(body) };
  const r = await fetch(path, opts);
  let data = {};
  try { data = await r.json(); } catch (e) { /* keep {} */ }
  return { status: r.status, data };
}

async function refresh() {
  try {
    const { data } = await api('/api/state');
    lastState = data;
    render();
  } catch (e) { /* server gone; keep last view */ }
}

function openCmd(wt, text, err) {
  const st = tabsFor(wt);
  const t = st.tabs.get('cmd') || { kind:'cmd', title:'command', buf:'', follow:false, err:false, meta:[] };
  t.buf = text; t.err = !!err;
  st.tabs.set('cmd', t);
  st.active = 'cmd';
  if (selected !== wt) selectWt(wt); else renderTabs();
}

async function act(cmd, wt) {
  busy.add(wt); render();
  openCmd(wt, '… running margay ' + cmd, false);
  try {
    const { data } = await api('/api/' + cmd, { worktreePath: wt });
    openCmd(wt, data.output || '(no output)', !data.ok);
  } catch (e) {
    openCmd(wt, 'request failed: ' + e, true);
  } finally {
    busy.delete(wt);
  }
  await refresh();
}

async function unregister(p) {
  try { await api('/api/unregister', { primaryPath: p }); } catch (e) { /* refresh shows reality */ }
  await refresh();
}

function actButton(wt, mini) {
  if (busy.has(wt.path)) return '<span class="spin">working…</span>';
  const cls = (wt.services.length ? 'down' : 'up') + (mini ? ' mini' : '');
  const cmd = wt.services.length ? 'down' : 'up';
  return '<button class="' + cls + '" onclick="event.stopPropagation(); act(\'' + cmd + '\',\'' + jsq(wt.path) + '\')">' + cmd + '</button>';
}

function renderSidebar() {
  const el = document.getElementById('sidebar');
  const ps = lastState.projects || [];
  if (!ps.length) {
    el.innerHTML = '<div class="empty">no projects yet — run <code>margay up</code> in a repo</div>';
    return;
  }
  el.innerHTML = ps.map(p => '<div class="proj-h" title="' + esc(p.primaryPath) + '"><b>' + esc(p.project) + '</b>' +
      (p.exists ? '' : ' <span class="badge">missing</span>') +
      '<button class="x" title="remove from this list (re-appears on next up)" ' +
      'onclick="unregister(\'' + jsq(p.primaryPath) + '\')">✕</button></div>' +
    (p.exists ? p.worktrees.map(wt =>
      '<div class="wt-row' + (wt.path === selected ? ' sel' : '') + '" onclick="selectWt(\'' + jsq(wt.path) + '\')">' +
      '<span class="dot' + (wt.services.length ? ' on' : '') + '"></span>' +
      '<b>' + esc(wt.isPrimary ? 'primary' : wt.name) + '</b>' +
      actButton(wt, true) + '</div>').join('') : '')).join('');
}

function renderSummary() {
  const el = document.getElementById('summary');
  const hit = selected && findWt(selected);
  if (!hit) { el.innerHTML = '<div class="empty">select a worktree</div>'; return; }
  const { wt } = hit;
  const rows = [];
  if (wt.url) rows.push(['root', '<a href="' + esc(wt.url) + '" target="_blank" rel="noopener">' + esc(wt.url) + '</a>']);
  else if (wt.hintUrl) rows.push(['root', '<span class="hint">' + esc(wt.hintUrl) + ' (not running)</span>']);
  for (const s of wt.services)
    rows.push([esc(s.service), '<a href="' + esc(s.url) + '" target="_blank" rel="noopener">' + esc(s.url) + '</a>']);
  const meta = wt.services.map(s =>
    esc(s.service) + ': pid ' + esc(s.pid) + (s.dbName ? ' · db ' + esc(s.dbName) : '')).join('  ·  ');
  el.innerHTML =
    '<h2>' + esc(wt.isPrimary ? 'primary' : wt.name) + ' ' + actButton(wt, false) + '</h2>' +
    '<div class="branch">' + esc(wt.branch) + '</div>' +
    '<div class="path">' + esc(wt.path) + '</div>' +
    (wt.collision ? '<div class="badge">slug collision — port links only</div>' : '') +
    (rows.length ? '<div class="urls">' + rows.map(r => '<span class="k">' + r[0] + '</span><span>' + r[1] + '</span>').join('') + '</div>' : '') +
    (meta ? '<div class="meta">' + meta + '</div>' : '');
}

function render() {
  if (!selected || !findWt(selected)) selected = defaultSelection();
  renderSidebar();
  renderSummary();
  const hit = selected && findWt(selected);
